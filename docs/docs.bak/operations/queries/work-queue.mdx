Get all work from the queue, by default sorted by start date (desc), priority (desc), originalWorkId (asc) and created (asc)

workQueue(
  limit: Int = 10
  offset: Int = 0
  status: [WorkStatus!]
  created: DateFilterInput
  queryString: String
  sort: [SortOptionInput!]
  types: [WorkType!]
): [Work!]!
